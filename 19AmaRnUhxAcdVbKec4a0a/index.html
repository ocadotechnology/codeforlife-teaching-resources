def next_turn(world_state, avatar_state):
    # scan_nearby() returns a list of the nearest artefacts
    nearby = world_state.scan_nearby(avatar_state.location)

    # Head towards the nearest artefact
    nearest = nearby[0]

    return MoveTowardsAction(nearest)

def next_turn(world_state, avatar_state):
    count = 0 # This is a local variable so it resets each call
    avatar_location = avatar_state.location
    current_cell = world_state.get_cell(avatar_location)

    if current_cell.has_artefact():
        count = count + 1
        print("I have", count, "things in my backpack")
        action = PickupAction()
    else:
        nearby = world_state.scan_nearby(avatar_location)
        nearest = nearby[0]
        action = MoveTowardsAction(nearest)

    return action
